1/**
2 * @file all_plots.c
3 * @brief Writers for receiver and satellite tracks used by plotting scripts.
4 *
5 * This module dumps computed GNSS artifacts to simple, columnar text files that
6 * gnuplot (or any plotting tool) can consume. It includes:
7 * - Receiver ECEF track (meters)
8 * - Receiver ECEF track with epoch index (kilometers)
9 * - Receiver geographic track (lat, lon in degrees)
10 * - Satellite orbits / ECEF samples (meters)
11 * - Satellite XYZ (kilometers)
12 * - (Optional) Pseudorange vs. time (kilometers)
13 *
14 * The routines are intentionally minimal and perform basic validation to avoid
15 * writing all-zero or non-finite rows.
16 */

28/**
29 * @brief Write receiver ECEF track (meters) to a file.
30 *
31 * Output format (per line): `X Y Z`
32 *
33 * @param path Destination file path.
34 * @param n_epochs Number of epochs to write.
35 * @return 0 on success, -1 on failure (open error or no lines written).
36 */
37int write_receiver_track_ecef(const char *path, int n_epochs)
38{
39 FILE *fp = fopen(path, "w");
40 if (!fp)
41 {
42 perror("[ERR] fopen(receiver_track)");
43 return -1;
44 }
45
46 int lines = 0;
47 for (int i = 0; i < n_epochs; ++i)
48 {
49 double x = estimated_positions_ecef.x[i];
50 double y = estimated_positions_ecef.y[i];
51 double z = estimated_positions_ecef.z[i];
52
53 /* Optional debug:
54 * printf("[DBG] epoch %d -> ECEF=(%.6f, %.6f, %.6f)\n", i, x, y, z);
55 */
56
57 /* Only write if not all zeros (safeguard) */
58 if (!(x == 0.0 && y == 0.0 && z == 0.0))
59 {
60 fprintf(fp, "%.8f %.8f %.8f\n", x, y, z);
61 lines++;
62 }
63 }
64
65 fclose(fp);
66
67 if (lines == 0)
68 {
69 fprintf(stderr, "[WARN] write_receiver_track_ecef: wrote 0 lines (all zero ECEF positions?)\n");
70 return -1;
71 }
72
73 return 0;
74}

76/**
77 * @brief Write satellite ECEF samples (meters) as one big file.
78 *
79 * Output format (per line): `PRN X Y Z`
80 * Sats are separated by blank lines.
81 *
82 * @param path Destination file path.
83 * @return 0 on success, -1 on failure to open.
84 */
85int write_sat_orbits(const char *path)
86{
87 FILE *fp = fopen(path, "w");
88 if (!fp)
89 return -1;
90
91 for (int prn = 1; prn <= MAX_SAT; ++prn)
92 {
93 int wrote_any = 0;
94 for (int k = 0; k < MAX_EPOCHS; ++k)
95 {
96 if (sat_ecef_positions[prn].t_ms[k] == 0.0)
97 continue;
98
99 double x = sat_ecef_positions[prn].x[k];
100 double y = sat_ecef_positions[prn].y[k];
101 double z = sat_ecef_positions[prn].z[k];
102
103 fprintf(fp, "%d %.6f %.6f %.6f\n", prn, x, y, z);
104 wrote_any = 1;
105 }
106 if (wrote_any)
107 fprintf(fp, "\n\n");
108 }
109
110 fclose(fp);
111 return 0;
112}

163/**
164 * @brief Write receiver ECEF (kilometers) with epoch index prefix.
165 *
166 * Output format (per line): `epoch_index X_km Y_km Z_km`
167 *
168 * @param path Destination file path.
169 * @param n_epochs Number of epochs to write.

172int write_receiver_ecef_epoch_km(const char *path, int n_epochs)
173{
174 FILE *fp = fopen(path, "w");
175 if (!fp)
176 return -1;
177
178 for (int i = 0; i < n_epochs; ++i)
179 {
180 double xk = m_to_km(estimated_positions_ecef.x[i]);
181 double yk = m_to_km(estimated_positions_ecef.y[i]);
182 double zk = m_to_km(estimated_positions_ecef.z[i]);
183
184 if (!isfinite(xk) || !isfinite(yk) || !isfinite(zk))
185 continue;
186
187 fprintf(fp, "%d %.6f %.6f %.6f\n", i, xk, yk, zk);
188 }
189
190 fclose(fp);
191 return 0;

235/**
236 * @brief (Optional) Write pseudorange vs. time per PRN (kilometers).
237 *
238 * Output format (per line): `PRN t(s) PR_km`
239 * Blocks are separated by blank lines.
240 *
241 * @param path Destination file path.
242 * @return 0 on success, -1 if file open failed.
243 *
244 * @note Time units: currently writes DF004/collected time as seconds if `t`
245 * is already in seconds. If your timestamps are in milliseconds, scale accordingly.
246 */
247int write_pseudorange_time_km(const char *path)
248{
249 FILE *fp = fopen(path, "w");
250 if (!fp)
251 return -1;
252
253 for (int prn = 1; prn <= MAX_SAT; ++prn)
254 {
255 for (int k = 0; k < MAX_EPOCHS; ++k)
256 {
257 uint32_t t = gps_list[prn].times_of_pseudorange[k];
258 double pr = gps_list[prn].pseudoranges[k]; /* meters */
259
260 if (t == 0 || !isfinite(pr))
261 continue;
262
263 /* Adjust if your t is milliseconds: use (double)t * 1e-3 */
264 double tow_s = (double)t;
265
266 fprintf(fp, "%d %.3f %.6f\n", prn, tow_s, pr * 1e-3);
267 }
268 fputs("\n\n", fp);
269 }
270
271 fclose(fp);
272 return 0;
