19// Helper: Find the closest eph index for prn whose EPH time_of_week is just before or equal to pseudorange_time
20static int find_closest_eph_idx(const eph_history_t *hist, uint32_t pseudorange_time)
21{
22 int best_idx = -1;
23 uint32_t best_toe = 0;
24 for (size_t i = 0; i < hist->count; i++)
25 {
26 uint32_t toe = hist->eph[i].gps_toe; // use TOE
27 if (toe <= pseudorange_time && toe >= best_toe)
28 {
29 best_toe = toe;
30 best_idx = (int)i;
31 }
32 }
33 return best_idx;
34}

36/* Populate gps_list[prn].{eccentricities,...,times_of_ephemeris} with the
37 FULL ephemeris history (unique by TOE), independent of pseudorange epochs. */
38static void populate_ephemeris_series_from_history(int prn, const eph_history_t *hist)
39{
40 // Reset ephemeris arrays
41 for (int k = 0; k < MAX_EPOCHS; k++)
42 {
43 gps_list[prn].eccentricities[k] = 0.0;
44 gps_list[prn].inclinations[k] = 0.0;
45 gps_list[prn].mean_anomalies[k] = 0.0;
46 gps_list[prn].semi_major_axes[k] = 0.0;
47 gps_list[prn].right_ascension_of_ascending_node[k] = 0.0;
48 gps_list[prn].argument_of_periapsis[k] = 0.0;
49 gps_list[prn].times_of_ephemeris[k] = 0;
50 }
51
52 size_t eidx = 0;
53 uint32_t last_toe = UINT32_MAX; // ensures first entry is accepted
54
55 for (size_t i = 0; i < hist->count && eidx < MAX_EPOCHS; i++)
56 {
57 const rtcm_1019_ephemeris_t *eph = &hist->eph[i];
58 uint32_t toe = eph->gps_toe;
59
60 // Append only when TOE changes (unique-by-TOE), mirroring Python's unique list
61 if (i == 0 || toe != last_toe)
62 {
63 gps_list[prn].eccentricities[eidx] = eph->eccentricity;
64 gps_list[prn].inclinations[eidx] = eph->inclination;
65 gps_list[prn].mean_anomalies[eidx] = eph->mean_anomaly;
66 gps_list[prn].semi_major_axes[eidx] = eph->semi_major_axis;
67 gps_list[prn].right_ascension_of_ascending_node[eidx] = eph->right_ascension_of_ascending_node;
68 gps_list[prn].argument_of_periapsis[eidx] = eph->argument_of_periapsis;
69 gps_list[prn].times_of_ephemeris[eidx] = toe;
70
71 last_toe = toe;
72 eidx++;
73 }
74 }
75}

77int sort_satellites(eph_history_t *eph_history_table,

79 rtcm_1002_msm1_t msm1_history_table[MAX_SAT][MAX_EPOCHS])

104 msm1_history_table[prn][i].time_of_week);

115 }

119 gps_list[prn].inclinations[i] = 0;
120 gps_list[prn].mean_anomalies[i] = 0;

125 }
126 break;
127 }
128 }
129 if (!found)
130 {
131 gps_list[prn].pseudoranges[i] = 0;
132 gps_list[prn].times_of_pseudorange[i] = 0;
133 gps_list[prn].eccentricities[i] = 0;
134 gps_list[prn].inclinations[i] = 0;
135 gps_list[prn].mean_anomalies[i] = 0;
136 gps_list[prn].semi_major_axes[i] = 0;
137 gps_list[prn].right_ascension_of_ascending_node[i] = 0;
138 gps_list[prn].argument_of_periapsis[i] = 0;
139 gps_list[prn].times_of_ephemeris[i] = 0;
140 }
141 }
142 }
143 }
144 else if (observation_type == 4)
145 {
146 for (int prn = 1; prn <= MAX_SAT; prn++)
147 {
148 gps_list[prn].prn = prn;
149
150 size_t msm4_len = msm4_count[prn];

156 {
157 if (msm4_history_table[prn][i].prn[j] == prn)

160 gps_list[prn].times_of_pseudorange[i] = msm4_history_table[prn][i].time_of_pseudorange;
161 found = 1;
162

164 int eph_idx = find_closest_eph_idx(&eph_history[prn],
165 msm4_history_table[prn][i].time_of_pseudorange);
166 if (eph_idx >= 0)
167 {
168 const rtcm_1019_ephemeris_t *eph = &eph_history_table[prn].eph[eph_idx];
169 gps_list[prn].eccentricities[i] = eph->eccentricity;
170 gps_list[prn].inclinations[i] = eph->inclination;
171 gps_list[prn].mean_anomalies[i] = eph->mean_anomaly;
172 gps_list[prn].semi_major_axes[i] = eph->semi_major_axis;
173 gps_list[prn].right_ascension_of_ascending_node[i] = eph->right_ascension_of_ascending_node;
174 gps_list[prn].argument_of_periapsis[i] = eph->argument_of_periapsis;

176 }

179 gps_list[prn].eccentricities[i] = 0;
180 gps_list[prn].inclinations[i] = 0;
181 gps_list[prn].mean_anomalies[i] = 0;
182 gps_list[prn].semi_major_axes[i] = 0;
183 gps_list[prn].right_ascension_of_ascending_node[i] = 0;
184 gps_list[prn].argument_of_periapsis[i] = 0;
185 gps_list[prn].times_of_ephemeris[i] = 0;
186 }
187 break;
188 }
189 }
190 if (!found)
191 {
192 gps_list[prn].pseudoranges[i] = 0;
193 gps_list[prn].times_of_pseudorange[i] = 0;
194 gps_list[prn].eccentricities[i] = 0;
195 gps_list[prn].inclinations[i] = 0;
196 gps_list[prn].mean_anomalies[i] = 0;
197 gps_list[prn].semi_major_axes[i] = 0;
198 gps_list[prn].right_ascension_of_ascending_node[i] = 0;
199 gps_list[prn].argument_of_periapsis[i] = 0;
200 gps_list[prn].times_of_ephemeris[i] = 0;
201 }
202 }
203 }
204 }
205 else
206 {
207 fprintf(stderr, COLOR_RED "Error: Unsupported observation type %d for sorting satellites.\n" COLOR_RESET, observation_type);
208 return 1;
209 }
210
211 // ---- Ephemeris history (independent of pseudoranges) — mirrors Python behavior ----
212 for (int prn = 1; prn <= MAX_SAT; prn++)
213 {
214 populate_ephemeris_series_from_history(prn, &eph_history[prn]);
215 }
216
217 return 0;
