1/**
2 * @file serial_connect.c
3 * @brief Enumerate and open a serial (COM/TTY) port on Windows or macOS/Linux.
4 *
5 * This module provides a small, interactive helper that lists available serial
6 * ports and opens the one the user selects. On Windows it searches COM ports
7 * via `QueryDosDeviceA()`. On macOS/Linux it scans `/dev` for `tty.usb*` and
8 * `ttyUSB*` device nodes and configures the port for 8-N-1 at 9600 baud.
9 *
10 * @note Logic is intentionally minimal and interactive; error handling favors
11 * clear console messages via `printf()`/`perror()`.
12 *
13 * @todo Make baud rate and serial settings configurable (currently hard-coded
14 * to 9600 8-N-1).
15 * @todo Add non-blocking / timeout configuration knobs.
16 * @todo Optionally accept a preselected port (skip interactive prompt).
17 */

28/**
29 * @brief List and connect to available COM ports on Windows.
30 *
31 * Enumerates COM1..COM256 using QueryDosDeviceA, prompts the user to choose a
32 * port, and attempts to open it with CreateFileA.
33 *
34 * @param[out] selected_port Buffer to receive the selected port name (e.g. "\\\\.\\COM3").
35 * @param[in] size Size of @p selected_port buffer in bytes.
36 * @return HANDLE to the opened serial port, or INVALID_HANDLE_VALUE on failure.
37 */
38HANDLE serial_connect_windows(char *selected_port, size_t size)
39{
40 char ports[256][16];
41 int count = 0;
42
43 printf("Available serial ports:\n");
44
45 for (int i = 1; i <= 256; i++)
46 {
47 char name[16];
48 char target[256];
49 DWORD size_check;
50 snprintf(name, sizeof(name), "COM%d", i);
51 size_check = QueryDosDeviceA(name, target, sizeof(target));
52 if (size_check > 0)
53 {
54 if (count >= (int)(sizeof(ports) / sizeof(ports[0])))
55 break;
56 snprintf(ports[count], sizeof(ports[0]), "%s", name);
57 printf(" [%d] %s\n", ++count, name);
58 }
59 }
60
61 if (count == 0)
62 {
63 printf("No serial ports found on Windows.\n");
64 return INVALID_HANDLE_VALUE;
65 }
66
67 int choice;
68 char input[16];
69 printf("Select a port by number (1-%d): ", count);
70 if (!fgets(input, sizeof(input), stdin) ||
71 sscanf(input, "%d", &choice) != 1 || choice < 1 || choice > count)
72 {
73 printf("Invalid selection.\n");
74 return INVALID_HANDLE_VALUE;
75 }
76
77 /* Prefix with \\.\ to allow COM ports > 9 */
78 snprintf(selected_port, size, "\\\\.\\%s", ports[choice - 1]);
79
80 HANDLE hSerial = CreateFileA(selected_port,
81 GENERIC_READ | GENERIC_WRITE,
82 0,
83 NULL,
84 OPEN_EXISTING,
85 0,
86 NULL);
87
88 if (hSerial == INVALID_HANDLE_VALUE)
89 {
90 DWORD err = GetLastError();
91 printf("Failed to open serial port (Error %lu)\n", err);
92 }
93 else
94 {
95 printf("Connected to %s\n", selected_port);
96 }
97
98 return hSerial;
99}

109/**
110 * @brief List and connect to USB serial ports on macOS/Linux.
111 *
112 * Scans `/dev` for `tty.usb*` and `ttyUSB*` device nodes, prompts for a choice,
113 * opens the device, and applies 9600 8-N-1 with basic raw settings.
114 *
115 * @param[out] selected_port Buffer to receive the selected device path.
116 * @param[in] size Size of @p selected_port buffer in bytes.
117 * @return File descriptor (>=0) on success, or -1 on failure.
118 */
119int serial_connect_mac(char *selected_port, size_t size)
120{
121 char ports[64][256];
122 size_t count = 0;
123
124 DIR *dp = opendir("/dev");
125 if (!dp)
126 {
127 perror("opendir /dev");
128 return -1;
129 }
130
131 printf("Available serial ports:\n");
132
133 struct dirent *entry;
134 while ((entry = readdir(dp)) != NULL)
135 {
136 if (fnmatch("tty.usb*", entry->d_name, 0) == 0 ||
137 fnmatch("ttyUSB*", entry->d_name, 0) == 0)
138 {
139 if (count >= (sizeof(ports) / sizeof(ports[0])))
140 break;
141 snprintf(ports[count], sizeof(ports[0]), "/dev/%s", entry->d_name);
142 printf(" [%zu] /dev/%s\n", count + 1, entry->d_name);
143 count++;
144 }
145 }
146
147 closedir(dp);
148
149 if (count == 0)
150 {
151 printf("No serial ports found on mac.\n");
152 return -1;
153 }
154
155 int choice;
156 char input[16];
157 printf("Select a port by number (1-%zu): ", count);
158 if (!fgets(input, sizeof(input), stdin) ||
159 sscanf(input, "%d", &choice) != 1 || choice < 1 || choice > (int)count)
160 {
161 printf("Invalid selection.\n");
162 return -1;
163 }
164
165 snprintf(selected_port, size, "%s", ports[choice - 1]);
166
167 int fd = open(selected_port, O_RDWR | O_NOCTTY | O_SYNC);
168 if (fd == -1)
169 {
170 perror("open serial port");
171 return -1;
172 }
173
174 struct termios tty;
175 if (tcgetattr(fd, &tty) != 0)
176 {
177 perror("tcgetattr");
178 close(fd);
179 return -1;
180 }
181
182 /* 9600 baud */
183 cfsetospeed(&tty, B9600);
184 cfsetispeed(&tty, B9600);
185
186 /* 8-N-1, no flow control */
187 tty.c_cflag &= (tcflag_t)(~CSIZE);
188 tty.c_cflag |= CS8;
189 tty.c_cflag &= (tcflag_t)(~PARENB);
190 tty.c_cflag &= (tcflag_t)(~CSTOPB);
191 tty.c_cflag &= (tcflag_t)(~CRTSCTS);
192 tty.c_cflag |= (CLOCAL | CREAD);
193
194 /* Raw-ish mode */
195 tty.c_lflag = 0;
196 tty.c_oflag = 0;
197 tty.c_iflag = 0;
198
199 /* Read returns as soon as 1 byte is available, with ~0.1s interbyte timeout */
200 tty.c_cc[VMIN] = 1;
201 tty.c_cc[VTIME] = 1;
202
203 if (tcsetattr(fd, TCSANOW, &tty) != 0)
204 {
205 perror("tcsetattr");
206 close(fd);
207 return -1;
208 }
209
210 printf("Connected to %s\n", selected_port);
211 return fd;
212}

215/**
216 * @brief Cross-platform serial connect wrapper.
217 *
218 * Dispatches to the platform-specific implementation and returns an opaque
219 * handle the caller can store. On Windows this is a `HANDLE`. On POSIX this is
220 * a file descriptor cast to `void*` via `intptr_t`.
221 *
222 * @param[out] selected_port Buffer to receive the chosen port path/name.
223 * @param[in] size Size of @p selected_port buffer in bytes.
224 * @return Opaque handle to the opened port, or NULL on failure.
225 */
226void *serial_connect(char *selected_port, size_t size)
227{
228#ifdef _WIN32
229 return (void *)serial_connect_windows(selected_port, size);
230#else
231 int fd = serial_connect_mac(selected_port, size);
232 return (fd >= 0) ? (void *)(intptr_t)fd : NULL;
233#endif
234}
