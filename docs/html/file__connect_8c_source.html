3 * @brief File input utility for GNSS Positioning Application.

5 * Provides user interaction for opening RTCM log files (either parsed text logs

7 * if no input is provided.

12/// Default file path used when user provides no input
13#define DEFAULT_FILE_PATH "example/parsed_log.txt"
14
15/// Maximum number of retry attempts for file input
16#define MAX_RETRIES 3
17
18/**
19 * @brief Open an RTCM log file for reading.
20 *
21 * Prompts the user to provide a path to an RTCM log file. If the input is left
22 * blank, a default path is used. Retries up to MAX_RETRIES if the file cannot
23 * be opened. Supports two modes:
24 * - Parsed mode (`is_parsed = true`): opens the file in text mode (`"r"`).
25 * - Raw mode (`is_parsed = false`): opens the file in binary mode (`"rb"`).
26 *
27 * @param[in] is_parsed True to open a parsed text log file, false for raw binary.
28 * @return FILE* Pointer to the opened file, or NULL on failure after all retries.
29 *
30 * @todo Implement raw binary RTCM parsing when `is_parsed = false`.
31 */
32FILE *file_connect(bool is_parsed)
33{
34 char file_path[256];
35 FILE *fp = NULL;
36
37 for (int attempt = 1; attempt <= MAX_RETRIES; ++attempt)
38 {
39 printf(COLOR_BLUE "\nEnter the path to the %s RTCM file,\n"
40 "or press Enter to use the default (%s):\n> " COLOR_RESET,
41 is_parsed ? "parsed text" : "raw binary",
42 DEFAULT_FILE_PATH);
43
44 // Get user input
45 if (fgets(file_path, sizeof(file_path), stdin) == NULL)
46 {
47 fprintf(stderr, COLOR_RED "Input error.\n" COLOR_RESET);
48 return NULL;
49 }
50
51 // Strip trailing newline
52 size_t len = strlen(file_path);
53 if (len > 0 && file_path[len - 1] == '\n')
54 file_path[len - 1] = '\0';
55
56 // If blank, use default path
57 if (strlen(file_path) == 0)
58 {
59 snprintf(file_path, sizeof(file_path), "%s", DEFAULT_FILE_PATH);
60 }
61
62 // Attempt to open file
63 if (is_parsed)
64 {
65 fp = fopen(file_path, "r"); ///< Parsed text log
66 }
67 else
68 {
69 fp = fopen(file_path, "rb"); ///< Raw binary log
70 // @todo Add logic for parsing raw RTCM binary streams
71 }
72
73 if (fp != NULL)
74 {
75 printf(COLOR_GREEN "Successfully opened file: %s\n" COLOR_RESET, file_path);
76 return fp;
77 }
78
79 // File open failed
80 fprintf(stderr, COLOR_RED "Error: Could not open file '%s': %s\n" COLOR_RESET,
81 file_path, strerror(errno));
82 printf(COLOR_YELLOW "Attempt %d of %d failed.\n" COLOR_RESET, attempt, MAX_RETRIES);
83 }
84
85 fprintf(stderr, COLOR_RED "Failed to open file after %d attempts. Exiting.\n" COLOR_RESET, MAX_RETRIES);
86 return NULL;
87}
