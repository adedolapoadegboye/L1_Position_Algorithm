1/**
2 * @file satellite_position_ecef.c
3 * @brief Converts the satellite position from ECI coordinates to ECEF frame.
4 *
5 * Mirrors the Python implementation:
6 * theta = (t / 86400) % 1 * 2*pi (solar day, not sidereal)
7 * and performs ecef = eci * Rz(theta).
8 * Since our helper does column-vector math (out = R * v),
9 * we multiply by Rz^T to match the Python row*matrix result.
10 */

22static inline double normalize_time_seconds(double t)
23{
24 // Many paths already store seconds, but some may still be ms.
25 // If t is suspiciously large, treat as milliseconds and convert.
26 // (E.g., 159348000 ms => 159348 s)
27 if (t > 1.0e6)
28 return t / 1000.0;
29 return t;
30}

32int satellite_position_ecef(const gps_satellite_data_t gps_lists[MAX_SAT + 1])
33{
34 for (int prn = 1; prn <= MAX_SAT; prn++)
35 {
36 for (int k = 0; k < MAX_EPOCHS; k++)
37 {
38 if (sat_eci_positions[prn].x[k] == 0.0 &&
39 sat_eci_positions[prn].y[k] == 0.0 &&
40 sat_eci_positions[prn].z[k] == 0.0)
41 {
42 continue;
43 }
44 else
45 {
46 // printf(s"PRN: %d, Orbit ECI[%d][0]: %.3f, Orbit ECI[%d][1]: %.3f, Orbit ECI[%d][2]: %.3f\n", prn, k, sat_eci_positions[prn].x[k] / 1000, k, sat_eci_positions[prn].y[k] / 1000, k, sat_eci_positions[prn].z[k] / 1000);
47 }
48
49 // Need a corresponding timestamp
50 double t_raw = gps_lists[prn].times_of_pseudorange[k];
51 if (t_raw == 0.0)
52 continue;
53
54 double t_sec = normalize_time_seconds(t_raw);
55 double frac_day = fmod(t_sec / 86400.0, 1.0);
56 if (frac_day < 0.0)
57 frac_day += 1.0;
58 double theta = frac_day * 2.0 * M_PI;
59
60 const double c = cos(theta), s = sin(theta);
61 const double Rz_T[3][3] = {
62 {c, s, 0.0},
63 {-s, c, 0.0},
64 {0.0, 0.0, 1.0}};
65
66 const double eci[3] = {
67 sat_eci_positions[prn].x[k],
68 sat_eci_positions[prn].y[k],
69 sat_eci_positions[prn].z[k]};
70
71 double ecef[3];
72 mat3x3_vec3_mult(Rz_T, eci, ecef);
73
74 sat_ecef_positions[prn].x[k] = ecef[0];
75 sat_ecef_positions[prn].y[k] = ecef[1];
76 sat_ecef_positions[prn].z[k] = ecef[2];
77 sat_ecef_positions[prn].t_ms[k] = t_sec * 1000.0; // store as ms
78
79 // Debug print
80 // printf("PRN %d, Epoch %d: t=%.3f s theta=%.6f rad ECI=[%.3f, %.3f, %.3f] ECEF=[%.3f, %.3f, %.3f]\n",
81 // prn, k, t_sec, theta, eci[0], eci[1], eci[2], ecef[0], ecef[1], ecef[2]);
82 }
83 }
84 return 0;
85}
