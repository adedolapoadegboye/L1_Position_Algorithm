1/**
2 * @file print_utils.c
3 * @brief Handles RTCM MSM4 file input, step-by-step parsing and position resolution.
4 *
5 * This function is triggered when the user selects "File Input" from the application menu.
6 * It opens a pre-recorded RTCM binary file, reads each epoch's MSM4 message, extracts relevant
7 * GNSS observations, and calculates the receiver position using the least squares method.
8 * The resulting position is printed in geodetic coordinates (decimal degrees).
9 *
10 * All major steps (reading, parsing, solving) are delegated to their respective modules
11 * to maintain modularity and educational clarity.
12 */

76/**
77 * @brief Prints the contents of a parsed RTCM 1002 (MSM1 GPS L1) observation.
78 *
79 * Uses the provided rtcm_1002_msm1_t fields (no assumptions).
80 * For each satellite, it also computes pseudorange with compute_pseudorange_msm1()
81 * from DF014 (ambiguity, ms) and DF011 (remainder, m) and shows the stored value.
82 */
83static const char *msm1_sig_name(uint8_t sig_id)
84{
85 // DF010: 00 = C/A, 01 = P(Y). Keep simple; extend as needed.
86 switch (sig_id)
87 {
88 case 0:
89 return "C/A";
90 case 1:
91 return "P(Y)";
92 default:
93 return "Unknown";
94 }
95}

97void print_msm1(const rtcm_1002_msm1_t *msm1)
98{
99 if (!msm1)
100 return;
101
102 printf("\n========= RTCM 1002 (MSM1 GPS L1) =========\n");
103
104 printf("Message Type : %u\n", msm1->msg_type);
105 printf("GPS TOW (s) : %u\n", msm1->time_of_week);
106 printf("Sync GNSS Msg Flag : %u\n", msm1->sync_gps_message_flag);
107 printf("Num Satellites (NSat) : %u\n", msm1->num_satellites);
108 printf("Smoothing Used? : %s\n", msm1->smooth_interval_flag ? "No divergence-free smoothing" : "Divergence-free smoothing");
109 printf("Smoothing Interval (s) : %u\n", msm1->smooth_interval);
110
111 // If any signal IDs were populated, show the distinct ones present.
112 // (Structure provides DF010 as an array; list any non-zero/known entries.)
113 bool printed_sig_header = false;
114 for (int s = 0; s < MAX_SIG; s++)
115 {
116 if (msm1->sig_id[s] || s == 0)
117 {
118 if (!printed_sig_header)
119 {
120 printf("\n-- Signal IDs Present (DF010) --\n");
121 printed_sig_header = true;
122 }
123 printf(" sig[%02d] = %u (%s)\n", s, msm1->sig_id[s], msm1_sig_name(msm1->sig_id[s]));
124 }
125 }
126
127 printf("\n-- Satellite Observations --\n");
128 printf("Idx PRN Amb(ms) Rem(m) PR(computed) PR(stored) Phase-PR(m) Lock C/N0(dBHz)\n");
129 printf("---- ---- ---------- ---------------- --------------------- --------------------- ------------------ ----- -----------\n");
130
131 for (int i = 0; i < msm1->num_satellites; i++)
132 {
133 uint8_t prn = msm1->svs[i];
134 double amb_ms = msm1->ambiguities[i]; // DF014 in ms
135 double rem_m = msm1->remainders[i]; // DF011 in m
136 double pr_comp = compute_pseudorange_msm1(amb_ms, rem_m);
137 double pr_stored = msm1->pseudoranges[i]; // already in meters
138 double phase_pr = msm1->phase_pr_diff[i]; // DF012 in meters
139 uint8_t lock = msm1->lock_time[i]; // DF013 in seconds
140 uint8_t cnr = msm1->cnr[i]; // DF015 (already scaled to dB-Hz)
141
142 printf("%-4d %-4u %-10.3f %-16.6f %-21.6f %-21.6f %-18.6f %-5u %-11u\n",
143 i + 1, prn, amb_ms, rem_m, pr_comp, pr_stored, phase_pr, lock, cnr);
144 }
145
146 printf("============================================\n");
147}

169 printf("Sync Flag : %u\n", msm4->msm_sync_flag);
170 printf("Clock Steering Flag : %u\n", msm4->clk_steering_flag);
171 printf("External Clock Flag : %u\n", msm4->external_clk_flag);
172 printf("Smoothing Flag : %u\n", msm4->smooth_interval_flag);
173
174 printf("Number of Satellites: %u\n", msm4->n_sat);
175 printf("Signal Type : %u\n", msm4->n_sig);
176 printf("Total Signal Count : %u\n", msm4->n_cell);
177
178 printf("\n-- Satellite PRNs --\n");
179 for (int i = 0; i < msm4->n_sat; i++)
180 {
181 printf(" PRN_%02d: %u | Integer PR: %u | Mod PR: %.12f | Fine PR: %.24f | Full PR: %.24f\n",
182 i + 1,
183 msm4->prn[i],
184 msm4->pseudorange_integer[i],
185 msm4->pseudorange_mod_1s[i],
186 msm4->pseudorange_fine[i],
187 msm4->pseudorange[i]);
188 }
189
190 printf("\n-- L1C Cell Observations --\n");
191 for (int i = 0; i < msm4->n_cell; i++)
192 {
193 printf(" Cell %02d:\n", i + 1);
194 printf(" PRN : %u\n", msm4->cell_prn[i]);
195 printf(" Signal ID : %u (1 = L1C)\n", msm4->cell_sig[i]);
196 printf(" Lock Time : %u\n", msm4->lock_time[i]);
197 printf(" CNR : %u dBHz\n", msm4->cnr[i]);
198 }
199
200 printf("========================================\n");
201}

205/**
206 * @brief Prints the stored pseudoranges for all satellites.
207 * This function iterates through the global pseudorange table
208 * and prints the details for each satellite that has valid ephemeris data.
209 */
210void print_all_stored_pseudoranges(void)
211{
212 for (int prn = 1; prn <= MAX_SAT; prn++)
213 {
214 if (pseudorange_count[prn] == 0)
215 continue;
216

219 printf("---------+--------------------\n");
220
221 for (size_t j = 0; j < pseudorange_count[prn]; j++)
222 {
223 printf("%-8zu | %-18.3f\n", j, pseudorange_history[prn][j]);
224 }
225 }

230/**
231 * @brief Prints the full ephemeris history for all PRNs.
232 *
233 * Iterates over all PRNs and prints every stored ephemeris entry from eph_history[].
234 */
235void print_all_stored_ephemeris(void)
236{
237 printf("\n========== Stored Ephemeris History ==========\n");
238
239 for (int prn = 1; prn <= MAX_SAT; prn++)
240 {
241 if (eph_history[prn].count == 0)
242 continue; // No entries for this PRN
243
244 printf("\nPRN %d: %zu entries stored\n", prn, eph_history[prn].count);
245
246 for (size_t idx = 0; idx < eph_history[prn].count; idx++)
247 {
248 const rtcm_1019_ephemeris_t *eph = &eph_history[prn].eph[idx];
249
250 printf(" [Epoch %zu]\n", idx);
251 printf(" Week Number: %u\n", eph->week_number);
252 printf(" Time of Week: %u s\n", eph->time_of_week);
253 printf(" Toe: %u s\n", eph->gps_toe);
254 printf(" Eccentricity: %.12e\n", eph->eccentricity);
255 printf(" Semi-Major Axis: %.3f m\n", eph->semi_major_axis);
256 printf(" Inclination: %.12e rad\n", eph->inclination);
257 printf(" RAAN: %.12e rad\n", eph->right_ascension_of_ascending_node);
258 printf(" Argument of Perigee: %.12e rad\n", eph->argument_of_periapsis);
259 printf(" Mean Anomaly: %.12e rad\n", eph->mean_anomaly);

265 printf("\n==============================================\n");
