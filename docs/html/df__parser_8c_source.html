10 * The input is assumed to be text-format (e.g., exported from parsed binary logs),

31/**
32 * @brief Parses a single RTCM 1002 MSM1 line into a structured observation object.
33 *
34 * This function extracts the MSM1 header and all per-satellite fields
35 * (PRN, signal ID, pseudorange remainder, phase range difference, lock time,
36 * ambiguity, and CNR) from the provided text-formatted RTCM message.
37 *
38 * @param line Input string containing the RTCM 1002 MSM1 message.
39 * @param msm1 Pointer to the MSM1 observation structure to populate.
40 * @return 0 on success, non-zero on failure.
41 */
42int parse_rtcm_1002(const char *line, rtcm_1002_msm1_t *msm1)
43{
44 if (!line || !msm1)
45 return -1;
46
47 EXTRACT("DF002", "%hu", &msm1->msg_type);
48 EXTRACT("DF003", "%hu", &msm1->station_id);
49 EXTRACT("DF004", "%u", &msm1->time_of_week);
50 msm1->time_of_week = msm1->time_of_week;
51 EXTRACT("DF005", "%hhu", &msm1->sync_gps_message_flag);
52 EXTRACT("DF006", "%hhu", &msm1->num_satellites);
53 EXTRACT("DF007", "%hhu", &msm1->smooth_interval_flag);
54 EXTRACT("DF008", "%hhu", &msm1->smooth_interval);
55
56 // Step X: Extract DF009_xx .. DF015_xx using strstr + sscanf
57 for (int i = 0; i < msm1->num_satellites; i++)
58 {
59 char key[16];
60 const char *ptr;
61
62 // DF009_xx: Satellite PRN
63 sprintf(key, "DF009_%02d", i + 1);
64 ptr = strstr(line, key);
65 if (ptr)
66 sscanf(ptr + strlen(key) + 1, "%hhu", &msm1->svs[i]);
67
68 // DF010_xx: Signal ID
69 sprintf(key, "DF010_%02d", i + 1);
70 ptr = strstr(line, key);
71 if (ptr)
72 sscanf(ptr + strlen(key) + 1, "%hhu", &msm1->sig_id[i]);
73
74 // DF011_xx: Pseudorange remainder (m)
75 sprintf(key, "DF011_%02d", i + 1);
76 ptr = strstr(line, key);
77 if (ptr)
78 sscanf(ptr + strlen(key) + 1, "%lf", &msm1->remainders[i]);
79
80 // DF012_xx: Carrier phase range minus pseudorange (m)
81 sprintf(key, "DF012_%02d", i + 1);
82 ptr = strstr(line, key);
83 if (ptr)
84 sscanf(ptr + strlen(key) + 1, "%lf", &msm1->phase_pr_diff[i]);
85
86 // DF013_xx: Lock time indicator
87 sprintf(key, "DF013_%02d", i + 1);
88 ptr = strstr(line, key);
89 if (ptr)
90 sscanf(ptr + strlen(key) + 1, "%hhu", &msm1->lock_time[i]);
91
92 // DF014_xx: Pseudorange modulus ambiguity
93 sprintf(key, "DF014_%02d", i + 1);
94 ptr = strstr(line, key);
95 if (ptr)
96 sscanf(ptr + strlen(key) + 1, "%hhu", &msm1->ambiguities[i]);
97
98 // DF015_xx: Carrier-to-noise ratio (dB-Hz)
99 sprintf(key, "DF015_%02d", i + 1);
100 ptr = strstr(line, key);
101 if (ptr)
102 sscanf(ptr + strlen(key) + 1, "%hhu", &msm1->cnr[i]);
103
104 // Calculate full pseudorange
105 msm1->pseudoranges[i] = compute_pseudorange_msm1(msm1->ambiguities[i], msm1->remainders[i]);
106 }
107 // print_msm1(msm1); // quick debug print
108 return 0;
109}

113/**
114 * @brief Parses a single RTCM 1074 MSM4 line into a structured observation object.
115 *
116 * This function extracts the MSM4 header and all per-cell data (pseudorange,
117 * carrier phase, lock time, CNR, etc.) for each satellite-signal combination.
118 *
119 * @param line Input string with the RTCM message content.
120 * @param msm4 Pointer to output structure to populate.
121 * @return 0 on success, non-zero on failure.
122 */
123int parse_rtcm_1074(const char *line, rtcm_1074_msm4_t *msm4)
124{
125 if (!line || !msm4)
126 return -1;
127
128 // Step 1: Extract MSM4 header fields using the EXTRACT macro
129 EXTRACT("DF002", "%hu", &msm4->msg_type);
130 EXTRACT("DF003", "%hu", &msm4->station_id);
131 EXTRACT("DF004", "%u", &msm4->gps_epoch_time);
132 msm4->time_of_pseudorange = msm4->gps_epoch_time;
133 EXTRACT("DF393", "%hhu", &msm4->msm_sync_flag);
134 EXTRACT("DF409", "%hhu", &msm4->iods_reserved);
135 EXTRACT("DF001_7", "%hhu", &msm4->reserved_DF001_07);
136 EXTRACT("DF411", "%hhu", &msm4->clk_steering_flag);
137 EXTRACT("DF412", "%hhu", &msm4->external_clk_flag);
138
139 // Step 2: Extract counts
140 EXTRACT("NSat", "%hhu", &msm4->n_sat);
141 EXTRACT("NSig", "%hhu", &msm4->n_sig);
142 EXTRACT("NCell", "%hhu", &msm4->n_cell);
143
144 // Step 3: Extract satellite PRNs (PRN_01, PRN_02, ...)
145 for (int i = 0; i < msm4->n_sat; i++)
146 {
147 char key[16];
148 sprintf(key, "PRN_%02d", i + 1);
149 const char *ptr = strstr(line, key);
150 if (ptr)
151 sscanf(ptr + strlen(key) + 1, "%hhu", &msm4->prn[i]);
152 }
153
154 // Step 4: Extract Pseudorange values
155 for (int i = 0; i < msm4->n_sat; i++)
156 {
157 char key[16];
158 sprintf(key, "DF397_%02d", i + 1);
159 const char *ptr = strstr(line, key);
160 if (ptr)
161 sscanf(ptr + strlen(key) + 1, "%hhu", &msm4->pseudorange_integer[i]);
162 }
163
164 for (int i = 0; i < msm4->n_sat; i++)
165 {
166 char key[16];
167 sprintf(key, "DF398_%02d", i + 1);
168 const char *ptr = strstr(line, key);
169 if (ptr)
170 sscanf(ptr + strlen(key) + 1, "%lf", &msm4->pseudorange_mod_1s[i]);
171 }
172
173 // Step 5: Extract Cell PRN and Signal ID mappings (CELLPRN_xx, CELLSIG_xx). Note: Store only L1 (1C) cells
174 uint8_t l1_cell_index = 0;
175
176 for (int i = 0; i < msm4->n_cell; i++)
177 {
178 char prn_key[24], sig_key[24], sig_val[8];
179 const char *ptr;
180
181 uint8_t prn_val = 0;
182
183 // Get CELLPRN
184 sprintf(prn_key, "CELLPRN_%02d", i + 1);
185 ptr = strstr(line, prn_key);
186 if (ptr)
187 sscanf(ptr + strlen(prn_key) + 1, "%hhu", &prn_val);
188
189 // Get CELLSIG
190 sprintf(sig_key, "CELLSIG_%02d", i + 1);
191 ptr = strstr(line, sig_key);
192 if (ptr)
193 {
194 sscanf(ptr + strlen(sig_key) + 1, "%2s", sig_val);
195
196 if (strcmp(sig_val, "1C") == 0)
197 {
198 // Keep only if signal is 1C
199 msm4->cell_prn[l1_cell_index] = prn_val;
200 msm4->cell_sig[l1_cell_index] = 1; // 1 = 1C in your system
201 l1_cell_index++;
202 }
203 }
204 }
205
206 // Step 6: Extract DF400 (pseudorange), DF401 (carrier phase), DF402 (lock), DF403 (CNR)
207 l1_cell_index = 0;
208
209 for (int i = 0; i < msm4->n_cell; i++)
210 {
211 char prn_key[24], sig_key[24], sig_val[8];
212 const char *ptr;
213 uint8_t prn_val = 0;
214
215 int cell_number = i + 1;
216
217 // CELLPRN_XX
218 sprintf(prn_key, "CELLPRN_%02d", cell_number);
219 ptr = strstr(line, prn_key);
220 if (ptr)
221 sscanf(ptr + strlen(prn_key) + 1, "%hhu", &prn_val);
222
223 // CELLSIG_XX
224 sprintf(sig_key, "CELLSIG_%02d", cell_number);
225 ptr = strstr(line, sig_key);
226 if (ptr)
227 {
228 sscanf(ptr + strlen(sig_key) + 1, "%2s", sig_val);
229
230 if (strcmp(sig_val, "1C") == 0)
231 {
232 // Store PRN and signal
233 msm4->cell_prn[l1_cell_index] = prn_val;
234 msm4->cell_sig[l1_cell_index] = 1; // L1C
235
236 // Extract corresponding DF400_XX to DF403_XX
237 char key[32];
238
239 sprintf(key, "DF400_%02d", cell_number);
240 ptr = strstr(line, key);
241 if (ptr)
242 sscanf(ptr + strlen(key) + 1, "%lf", &msm4->pseudorange_fine[l1_cell_index]);
243
244 sprintf(key, "DF401_%02d", cell_number);
245 ptr = strstr(line, key);
246 if (ptr)
247 sscanf(ptr + strlen(key) + 1, "%lf", &msm4->phase_range[l1_cell_index]);
248
249 sprintf(key, "DF402_%02d", cell_number);
250 ptr = strstr(line, key);
251 if (ptr)
252 sscanf(ptr + strlen(key) + 1, "%hhu", &msm4->lock_time[l1_cell_index]);
253
254 sprintf(key, "DF403_%02d", cell_number);
255 ptr = strstr(line, key);
256 if (ptr)
257 sscanf(ptr + strlen(key) + 1, "%hhu", &msm4->cnr[l1_cell_index]);
258
259 l1_cell_index++;
260 }
261 }
262 }
263
264 // Finalize the number of cells to the actual count of L1C cells
265 msm4->n_cell = l1_cell_index;
266
267 // Step 7: Calculate pseudorange array
268 for (int i = 0; i < msm4->n_cell; i++)
269 {
270 // Check if the cell is valid (has a PRN and signal)
271 if (msm4->cell_prn[i] > 0 && msm4->cell_sig[i] == 1) // Only L1C cells
272 {
273 msm4->pseudorange[i] = compute_pseudorange(
274 msm4->pseudorange_integer[i],
275 msm4->pseudorange_mod_1s[i],
276 msm4->pseudorange_fine[i]);
277 }
278 else
279 {
280 msm4->pseudorange[i] = -1.0; // mark as invalid
281 }
282 }
283
284 // print_msm4(msm4); // quick debug print
285
286 return 0;
287}

354/**
355 * @brief Computes the pseudorange from the given parameters.
356 *
357 * This function calculates the pseudorange using the formula:
358 * Pseudorange = c * (integer_ms * 1e-3 + mod1s_sec + fine_sec)
359 *
360 * @param integer_ms Rough range integer in milliseconds.
361 * @param mod1s_sec Pseudorange modulo 1 second.
362 * @param fine_sec Pseudorange residuals in seconds scaled by speed of light.
363 * @return The computed pseudorange in meters.
364 */
365double compute_pseudorange(uint32_t integer_ms, double mod1s_sec, double fine_sec)
366{
367 return SPEED_OF_LIGHT * (integer_ms * 1e-3) + mod1s_sec + fine_sec;
368}
369
370/**
371 * @brief Computes the pseudorange for an RTCM 1002 (MSM1) observation.
372 *
373 * Calculates the pseudorange by combining the rough range integer (in milliseconds)
374 * with the pseudorange remainder (in meters), applying the appropriate speed-of-light
375 * scaling to the integer term.
376 *
377 * Formula:
378 * pseudorange = (amb * 299792.458) + rem
379 *
380 * @param amb Rough range integer in milliseconds.
381 * @param rem Pseudorange remainder in meters.
382 * @return Pseudorange in meters.
383 */
384double compute_pseudorange_msm1(double amb, double rem)
385{
386 // Convert ambiguity from milliseconds to meters
387 double amb_meters = amb * (SPEED_OF_LIGHT / 1000.0);
388
389 // Sum with remainder to get full pseudorange
390 return (double)(amb_meters + rem);
391}

395/**
396 * @brief Stores or updates the ephemeris data for a given satellite.
397 *
398 * This function always updates the ephemeris for the specified satellite,
399 * regardless of IODE/IODC. It ensures the latest data is always stored.
400 *
401 * @param new_eph Pointer to the new ephemeris data to store.
402 * @return 0 on success, -1 if input is NULL, -2 if PRN is out of range (1-32).
403 */
404int store_ephemeris(const rtcm_1019_ephemeris_t *new_eph)
405{
406 if (!new_eph)
407 return -1;
408
409 uint8_t prn = new_eph->satellite_id;
410 if (prn < 1 || prn > MAX_SAT)
411 return -2;
412
413 size_t idx = eph_history[prn].count++;
414 if (idx < MAX_EPH_HISTORY)
415 {
416 eph_history[prn].eph[idx] = *new_eph;
417 // printf(COLOR_GREEN "Stored ephemeris for PRN %u at idx %zu\n" COLOR_RESET, prn, idx);
418 }
419 else
420 {
421 // printf(COLOR_RED "Warning: Ephemeris history for PRN %u exceeded max entries (%d). Data may be lost.\n" COLOR_RESET, prn, MAX_EPH_HISTORY);
422 }
423
424 // Optionally update eph_table[prn] and eph_available[prn] for legacy code
425 eph_table[prn] = *new_eph;
426 eph_available[prn] = true;
427
428 return 0;
429}
