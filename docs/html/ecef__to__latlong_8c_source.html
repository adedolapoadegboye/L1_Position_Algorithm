1/**
2 * @file ecef_to_geodetic.c
3 * @brief Convert ECEF (Earth-Centered, Earth-Fixed) coordinates to geodetic coordinates (WGS-84).
4 *
5 * This function converts Earth-Centered Earth-Fixed (ECEF) Cartesian coordinates
6 * (X, Y, Z in meters) into geodetic latitude, longitude, and altitude above the WGS-84 ellipsoid.
7 *
8 * Implementation details:
9 * - Uses WGS-84 constants (semi-major axis and flattening).
10 * - Applies Bowring’s formula for latitude computation.
11 * - Computes ellipsoidal height (altitude) relative to the WGS-84 reference ellipsoid.
12 *
13 * @param[in] x ECEF X coordinate (meters).
14 * @param[in] y ECEF Y coordinate (meters).
15 * @param[in] z ECEF Z coordinate (meters).
16 * @param[out] lat_deg Geodetic latitude in degrees.
17 * @param[out] lon_deg Geodetic longitude in degrees.
18 * @param[out] alt_m Altitude above WGS-84 ellipsoid (meters).
19 *
20 * @note If all inputs (x, y, z) are zero, the function returns lat=0, lon=0, alt=-a (arbitrary).
21 */

25void ecef_to_geodetic(double x, double y, double z,
26 double *lat_deg, double *lon_deg, double *alt_m)
27{
28 // WGS-84 constants
29 const double a = 6378137.0; // semi-major axis (m)
30 const double f = 1.0 / 298.257223563; // flattening
31 const double b = a * (1.0 - f); // semi-minor axis (m)
32 const double e2 = 2.0 * f - f * f; // first eccentricity^2
33 const double ep2 = (a * a - b * b) / (b * b); // second eccentricity^2
34 const double rad2deg = 180.0 / PI;
35
36 // Longitude
37 double lon = atan2(y, x);
38
39 // Distance from Z axis
40 double p = sqrt(x * x + y * y);
41
42 // Guard against origin
43 if (p == 0.0 && z == 0.0)
44 {
45 if (lat_deg)
46 *lat_deg = 0.0;
47 if (lon_deg)
48 *lon_deg = 0.0;
49 if (alt_m)
50 *alt_m = -a; // arbitrary
51 return;
52 }
53
54 // Bowring’s formula for latitude
55 double theta = atan2(z * a, p * b);
56 double st = sin(theta), ct = cos(theta);
57 double lat = atan2(z + ep2 * b * st * st * st,
58 p - e2 * a * ct * ct * ct);
59
60 // Radius of curvature in the prime vertical
61 double sl = sin(lat);
62 double N = a / sqrt(1.0 - e2 * sl * sl);
63
64 // Altitude
65 double alt = (p / cos(lat)) - N;
66
67 if (lat_deg)
68 *lat_deg = lat * rad2deg;
69 if (lon_deg)
70 *lon_deg = lon * rad2deg;
71 if (alt_m)
72 *alt_m = alt;
73}
