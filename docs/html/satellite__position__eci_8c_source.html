24// Helper: Find the closest eph index for prn whose EPH time_of_week is just before or equal to pseudorange_time
25static int find_closest_eph_idx(const eph_history_t *hist, uint32_t pseudorange_time)
26{
27 int best_idx = -1;
28 uint32_t best_toe = 0;
29 for (size_t i = 0; i < hist->count; i++)
30 {
31 uint32_t toe = hist->eph[i].gps_toe;
32 if ((toe >= pseudorange_time || toe <= pseudorange_time) && toe >= best_toe)
33 {
34 best_toe = toe;
35 best_idx = (int)i;
36 }
37 }
38 return best_idx;
39}

41int satellite_position_eci(const gps_satellite_data_t gps_lists[])
42{
43 for (int prn = 1; prn <= MAX_SAT; prn++)
44 {
45 // Skip PRNs with no ephemeris history at all
46 if (eph_history[prn].count == 0)
47 {
48 // printf("No ephemeris history for PRN %d\n", prn);
49 continue;
50 }
51
52 for (int k = 0; k < MAX_EPOCHS; k++)
53 {
54 if (gps_lists[prn].times_of_pseudorange[k] == 0)
55 continue;
56
57 // --- 1) Normalize observation time to seconds ---
58 double t_obs = gps_lists[prn].times_of_pseudorange[k];
59
60 if (t_obs > 604800.0)
61 t_obs *= 1.0 / 1000.0;
62
63 // --- 2) Find best ephemeris for this observation time (TOE <= t_obs) ---
64 int eph_idx = find_closest_eph_idx(&eph_history[prn], (uint32_t)t_obs);
65 if (eph_idx < 0)
66 {
67 // No suitable ephemeris before/at this time; skip
68 // printf("No suitable ephemeris for PRN %d at t=%.3f\n", prn, t_obs);
69 continue;
70 }
71
72 const rtcm_1019_ephemeris_t *eph = &eph_history[prn].eph[eph_idx];
73
74 // Pull elements from ephemeris (radians & meters as you stored them)
75 const double a = eph->semi_major_axis;
76 const double e = eph->eccentricity;
77 const double i = eph->inclination;
78 const double Omega = eph->right_ascension_of_ascending_node;
79 const double omega = eph->argument_of_periapsis;
80 const double M0 = eph->mean_anomaly;
81 const double toe = (double)eph->gps_toe; // seconds
82
83 // Basic sanity guards
84 if (!(a > 0.0) || !(e >= 0.0 && e < 1.0) || !isfinite(i) || !isfinite(M0))
85 {
86 // printf("Bad elements for PRN %d at k=%d (a=%g, e=%g)\n", prn, k, a, e);
87 continue;
88 }
89
90 // --- 3) Time since ephemeris epoch ---
91 const double dt = t_obs - toe;
92 // printf("PRN %d, Epoch %d: dt = %.3f seconds\n", prn, k, dt);
93
94 // --- 4) Mean motion (rad/s) and mean anomaly ---
95 const double n = sqrt(MU / (a * a * a)); // MU in m^3/s^2
96 double M = M0 + n * dt;
97
98 // Normalize M into [-pi, pi] for numerical stability
99 M = fmod(M + M_PI, 2.0 * M_PI);
100 if (M < 0)
101 M += 2.0 * M_PI;
102 M -= M_PI;
103
104 // --- 5) Solve Kepler for E (eccentric anomaly) via simple iteration (good enough for GPS e) ---
105 double E = M; // initial guess
106 for (int it = 0; it < 10; ++it)
107 {
108 double f = E - e * sin(E) - M;
109 double fp = 1.0 - e * cos(E);
110 double dE = -f / fp;
111 E += dE;
112 if (fabs(dE) < 1e-12)
113 break;
114 }
115
116 // True anomaly
117 double cosE = cos(E), sinE = sin(E);
118 double sqrt1me2 = sqrt(fmax(0.0, 1.0 - e * e));
119 double sinv = (sqrt1me2 * sinE) / (1.0 - e * cosE);
120 double cosv = (cosE - e) / (1.0 - e * cosE);
121 double v = atan2(sinv, cosv);
122 // printf("PRN %d, Epoch %d: True Anomaly = %.6f\n", prn, k, v);
123
124 // Radius in the orbital plane
125 double r = a * (1.0 - e * cosE);
126 if (!(r > 0.0) || !isfinite(r))
127 {
128 // printf("Bad radius for PRN %d (r=%g)\n", prn, r);
129 continue;
130 }
131
132 // PQW coordinates (perifocal frame)
133 double pqw[3] = {r * cos(v), r * sin(v), 0.0};
134
135 // printf("PRN %d, Epoch %d: PQW = [%.3f, %.3f, %.3f]\n", prn, k, pqw[0], pqw[1], pqw[2]);
136
137 // --- 6) Rotation matrices to ECI: Rz(Ω) * Rx(i) * Rz(ω) ---
138 // We rotate +ω about z, +i about x, +Ω about z (no leading negatives needed)
139 const double cO = cos(Omega), sO = sin(Omega);
140 const double ci = cos(i), si = sin(i);
141 const double co = cos(omega), so = sin(omega);
142
143 // Rz(ω)
144 double Rz_omega[3][3] = {
145 {co, -so, 0},
146 {so, co, 0},
147 {0, 0, 1}};
148 // Rx(i)
149 double Rx_i[3][3] = {
150 {1, 0, 0},
151 {0, ci, -si},
152 {0, si, ci}};
153 // Rz(Ω)
154 double Rz_Omega[3][3] = {
155 {cO, -sO, 0},
156 {sO, cO, 0},
157 {0, 0, 1}};
158
159 // Apply rotations: ECI = Rz(Ω) * Rx(i) * Rz(ω) * pqw
160 double tmp1[3], tmp2[3], eci[3];
161 mat3x3_vec3_mult(Rz_omega, pqw, tmp1);
162 mat3x3_vec3_mult(Rx_i, tmp1, tmp2);
163 mat3x3_vec3_mult(Rz_Omega, tmp2, eci);
164
165 // Save
166 sat_eci_positions[prn].x[k] = eci[0];
167 sat_eci_positions[prn].y[k] = eci[1];
168 sat_eci_positions[prn].z[k] = eci[2];
169
170 // printf("PRN %d, Epoch %d: ECI = [%.3f, %.3f, %.3f]\n", prn, k, sat_eci_positions[prn].x[k] / 1000, sat_eci_positions[prn].y[k] / 1000, sat_eci_positions[prn].z[k] / 1000);
171 }
172 }
173 return 0;
174}
