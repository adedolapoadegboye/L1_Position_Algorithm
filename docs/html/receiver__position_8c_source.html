1/**
2 * @file receiver_position.c
3 * @brief Estimate GNSS receiver positions using pseudorange measurements.
4 *
5 * This module implements the C equivalent of the Python `estimate_positions`
6 * function. It:
7 * - Collects unique pseudorange epochs across all satellites
8 * - Aligns satellite ECEF positions to the pseudorange epochs
9 * - Runs an iterative least-squares solver (Newton method) to estimate
10 * receiver position and clock bias per epoch
11 *
12 * The output is stored in `estimated_positions_ecef` as ECEF coordinates.
13 *
14 * @note This implementation uses normal equations with a 4x4 matrix inverse
15 * for the pseudoinverse step. It mirrors the Python logic, with optional
16 * diagnostic prints disabled by default.
17 *
18 * @author Ade
19 * @date 2025-08-23
20 */

39static int cmp_u32(const void *a, const void *b)
40{
41 uint32_t A = *(const uint32_t *)a, B = *(const uint32_t *)b;
42 return (A > B) - (A < B);
43}

58static int collect_unique_pr_times_ms(const gps_satellite_data_t glist[MAX_SAT + 1],
59 uint32_t out_times_ms[MAX_UNIQUE_EPOCHS])
60{
61 size_t n = 0;
62
63 for (int prn = 1; prn <= MAX_SAT; prn++)
64 {
65 int n_pr = pr_count_for_prn(prn);
66 for (int k = 0; k < n_pr; k++)
67 {
68 uint32_t t = glist[prn].times_of_pseudorange[k];
69 if (t == 0)
70 continue;
71 if (n >= (size_t)MAX_UNIQUE_EPOCHS)
72 goto SORT_AND_DEDUP;
73 out_times_ms[n++] = t;
74 }
75 }
76
77SORT_AND_DEDUP:
78 if (n == 0)
79 return 0;
80
81 qsort(out_times_ms, n, sizeof(uint32_t), cmp_u32);
82
83 size_t w = 1;
84 for (size_t i = 1; i < n; i++)
85 {
86 if (out_times_ms[i] != out_times_ms[w - 1])
87 out_times_ms[w++] = out_times_ms[i];
88 }
89 return (int)w;
90}

92static inline double norm3(const double v[3])
93{
94 return sqrt(v[0] * v[0] + v[1] * v[1] + v[2] * v[2]);
95}

97/* 4x4 inverse (Gauss–Jordan, partial pivoting) */
98static int invert_4x4(const double A[4][4], double inv[4][4])
99{
100 double aug[4][8];
101 for (int r = 0; r < 4; ++r)
102 {
103 for (int c = 0; c < 4; ++c)
104 aug[r][c] = A[r][c];
105 for (int c = 0; c < 4; ++c)
106 aug[r][4 + c] = (r == c) ? 1.0 : 0.0;
107 }
108 for (int col = 0; col < 4; ++col)
109 {
110 int piv = col;
111 double maxabs = fabs(aug[piv][col]);
112 for (int r = col + 1; r < 4; ++r)
113 {
114 double v = fabs(aug[r][col]);
115 if (v > maxabs)
116 {
117 maxabs = v;
118 piv = r;
119 }
120 }
121 if (maxabs <= 1e-18)
122 return 0;
123 if (piv != col)
124 {
125 for (int c = 0; c < 8; ++c)
126 {
127 double tmp = aug[col][c];
128 aug[col][c] = aug[piv][c];
129 aug[piv][c] = tmp;
130 }
131 }
132 double invpiv = 1.0 / aug[col][col];
133 for (int c = 0; c < 8; ++c)
134 aug[col][c] *= invpiv;
135 for (int r = 0; r < 4; ++r)
136 {
137 if (r == col)
138 continue;
139 double f = aug[r][col];
140 if (f != 0.0)
141 for (int c = 0; c < 8; ++c)
142 aug[r][c] -= f * aug[col][c];
143 }
144 }
145 for (int r = 0; r < 4; ++r)
146 for (int c = 0; c < 4; ++c)
147 inv[r][c] = aug[r][4 + c];
148 return 1;
149}

151/* delta(4) = pinv(G)(4xm) * y(m) via normal equations */
152static int pinv_normal_eq_apply(int m, /* rows of G */
153 double G[][4], /* m x 4 */
154 const double *y, /* m x 1 */
155 double out[4]) /* 4 x 1 */
156{
157 /* Compute AT = G^T (4 x m) and ATA = G^T G (4x4), ATy = G^T y (4) */
158 double AT[4][MAX_SAT];
159 for (int i = 0; i < 4; ++i)
160 for (int j = 0; j < m; ++j)
161 AT[i][j] = G[j][i];
162
163 double ATA[4][4] = {{0}};
164 for (int r = 0; r < 4; ++r)
165 for (int c = 0; c < 4; ++c)
166 {
167 double acc = 0.0;
168 for (int k = 0; k < m; ++k)
169 acc += AT[r][k] * G[k][c];
170 ATA[r][c] = acc;
171 }
172
173 double ATy[4] = {0, 0, 0, 0};
174 for (int r = 0; r < 4; ++r)
175 {
176 double acc = 0.0;
177 for (int k = 0; k < m; ++k)
178 acc += AT[r][k] * y[k];
179 ATy[r] = acc;
180 }
181
182 double invATA[4][4];
183 if (!invert_4x4(ATA, invATA))
184 return 0;
185
186 for (int r = 0; r < 4; ++r)
187 {
188 double acc = 0.0;
189 for (int c = 0; c < 4; ++c)
190 acc += invATA[r][c] * ATy[c];
191 out[r] = acc;
192 }
193 return 1;
194}

196/* ---------- main function ---------- */
197int estimate_receiver_positions(void)
198{
199 /* 1) Epoch collection (np.unique over all PR times) */
200 uint32_t *all_times = (uint32_t *)malloc(sizeof(uint32_t) * (size_t)MAX_UNIQUE_EPOCHS);
201 if (!all_times)
202 {
203 perror("malloc(all_times)");
204 return -1;
205 }
206 for (int i = 0; i < MAX_UNIQUE_EPOCHS; ++i)
207 all_times[i] = 0;
208
209 n_times = collect_unique_pr_times_ms(gps_list, all_times);
210 printf("[C] total epochs = %d\n", n_times);
211 if (n_times > 0 && n_times <= 20)
212 {
213 printf("[C] epochs (ms): ");
214 for (int i = 0; i < n_times; ++i)
215 printf("%u%s", all_times[i], (i + 1 < n_times) ? ", " : "\n");
216 }
217
218 /* 2) Per-SV summary (PR samples, ECEF shape, first/last PR time) */
219 for (int prn = 1; prn <= MAX_SAT; ++prn)
220 {
221 int pr_cnt = pr_count_for_prn(prn);
222 int ecef_rows = 0;
223 for (int k = 0; k < MAX_EPOCHS; ++k)
224 if (sat_ecef_positions[prn].t_ms[k] != 0.0)
225 ecef_rows++;
226
227 if (pr_cnt > 0)
228 {
229 /* first/last PR time within recorded range */
230 int first_idx = -1, last_idx = -1;
231 for (int k = 0; k < MAX_EPOCHS; ++k)
232 {
233 if (gps_list[prn].times_of_pseudorange[k] != 0)
234 {
235 first_idx = k;
236 break;
237 }
238 }
239 for (int k = MAX_EPOCHS - 1; k >= 0; --k)
240 {
241 if (gps_list[prn].times_of_pseudorange[k] != 0)
242 {
243 last_idx = k;
244 break;
245 }
246 }
247 uint32_t first_t = (first_idx >= 0) ? gps_list[prn].times_of_pseudorange[first_idx] : 0u;
248 uint32_t last_t = (last_idx >= 0) ? gps_list[prn].times_of_pseudorange[last_idx] : 0u;
249
250 printf("[C] SV %02d: PR samples=%d, ECEF shape=(%d,3); first PR time=%u, last PR time=%u\n",
251 prn, pr_cnt, ecef_rows, first_t, last_t);
252 }
253 else
254 {
255 // printf("[C] SV %02d: ECEF shape=(%d,3)\n", prn, ecef_rows);
256 }
257 }
258
259 /* 3) Process each epoch independently */
260 for (int ti = 0; ti < n_times; ++ti)
261 {
262 uint32_t t = all_times[ti];
263
264 int svs[MAX_SAT];
265 int pr_indices[MAX_SAT];
266 double ecefs[MAX_SAT][3];
267 double pseudoranges[MAX_SAT];
268 int n_svs = 0;
269
270 /* Gather same-time measurements (first match per SV) */
271 for (int prn = 1; prn <= MAX_SAT && n_svs < MAX_SAT; ++prn)
272 {
273 int n_pr = pr_count_for_prn(prn);
274 for (int k = 0; k < n_pr; ++k)
275 {
276 if (gps_list[prn].times_of_pseudorange[k] != t)
277 continue;
278
279 svs[n_svs] = prn;
280 pr_indices[n_svs] = k;
281 ecefs[n_svs][0] = sat_ecef_positions[prn].x[k];
282 ecefs[n_svs][1] = sat_ecef_positions[prn].y[k];
283 ecefs[n_svs][2] = sat_ecef_positions[prn].z[k];
284 pseudoranges[n_svs] = gps_list[prn].pseudoranges[k];
285 n_svs++;
286 break;
287 }
288 }
289
290 if (n_svs < 4)
291 continue;
292
293 /* --- Iterative least-squares (Newton) --- */
294 double assumed_pos[3] = {0.0, 0.0, 0.0};
295 double clock_bias = 0.0;
296
297 for (int it = 0; it < ITERATIONS; ++it)
298 {
299 double assumed_ranges[MAX_SAT];
300 double unit_vectors[MAX_SAT][3];
301 double delta_tau[MAX_SAT];
302
303 for (int i = 0; i < n_svs; ++i)
304 {
305 double los[3] = {
306 ecefs[i][0] - assumed_pos[0],
307 ecefs[i][1] - assumed_pos[1],
308 ecefs[i][2] - assumed_pos[2]};
309 double r = norm3(los);
310 if (!(r > 0.0) || !isfinite(r))
311 r = 1.0;
312
313 assumed_ranges[i] = r;
314 unit_vectors[i][0] = los[0] / r;
315 unit_vectors[i][1] = los[1] / r;
316 unit_vectors[i][2] = los[2] / r;
317
318 delta_tau[i] = pseudoranges[i] - r - clock_bias;
319 }
320
321 /* Geometry matrix G (n_svs x 4) */
322 double G[MAX_SAT][4];
323 for (int i = 0; i < n_svs; ++i)
324 {
325 G[i][0] = -unit_vectors[i][0];
326 G[i][1] = -unit_vectors[i][1];
327 G[i][2] = -unit_vectors[i][2];
328 G[i][3] = 1.0;
329 }
330
331 /* delta = pinv(G) * delta_tau (4x1) */
332 double delta_pos_time[4];
333 if (!pinv_normal_eq_apply(n_svs, G, delta_tau, delta_pos_time))
334 {
335 /* singular / ill-conditioned; skip this epoch */
336 n_svs = 0; /* to avoid storing a bogus result */
337 break;
338 }
339
340 assumed_pos[0] += delta_pos_time[0];
341 assumed_pos[1] += delta_pos_time[1];
342 assumed_pos[2] += delta_pos_time[2];
343 clock_bias += delta_pos_time[3];
344
345 // Optional prints:
346 double rms = 0.0;
347 for (int i = 0; i < n_svs; ++i)
348 rms += delta_tau[i] * delta_tau[i];
349 rms = sqrt(rms / n_svs);
350 // printf("[C][epoch %d][iter %d] rms=%.3f, clk=%.6f\n", ti, it, rms, clock_bias);
351 }
352
353 if (n_svs >= 4)
354 {
355 /* store final ECEF estimate (per epoch index ti) */
356 estimated_positions_ecef.x[ti] = assumed_pos[0];
357 estimated_positions_ecef.y[ti] = assumed_pos[1];
358 estimated_positions_ecef.z[ti] = assumed_pos[2];
359 // printf("[C][epoch %d] FINAL pos ECEF = (%.3f, %.3f, %.3f) m, clock_bias=%.6f m\n",
360 // ti, estimated_positions_ecef.x[ti], estimated_positions_ecef.x[ti], estimated_positions_ecef.x[ti], clock_bias);
361
362 /* --- Convert to geodetic and store --- */
363 if (ti < MAX_EPOCHS)
364 {
365 double lat_deg, lon_deg, alt_m;
366 ecef_to_geodetic(assumed_pos[0], assumed_pos[1], assumed_pos[2],
367 &lat_deg, &lon_deg, &alt_m);
368
369 /* single instance: store by epoch index */
370 latlonalt_positions.lat[ti] = lat_deg;
371 latlonalt_positions.lon[ti] = lon_deg;
372
373 /* optional print (comment out if noisy) */
374 printf("[C][epoch %d] LLA = (lat=%.8f deg, lon=%.8f deg)\n",
375 ti, lat_deg, lon_deg);
376 }
377 }
378 }
379
380 free(all_times);
381 return 0;
382}
